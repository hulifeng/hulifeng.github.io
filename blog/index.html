# 2019-02-13

# VuePress 学习

# 1. 安装

# 1.1 依赖环境：node8.0以上的版本

$ node -v
v10.5.0

# 1.2 安装VuePress

$ npm install -g vuepress

# 1.3 查看vuepress安装位置和版本

$ where vuepress
D:\node\node_global\vuepress
D:\node\node_global\vuepress.cmd

$ vuepress --version
0.14.8

# 2. 创建和安装

# 2.1 创建一个文件夹，并创建 README.md 文件

$ mkdir press
$ cd press
$ echo '# Hello Vuepress' > README.md

# 2.2 启动并测试

$ vuepress dev

测试： 浏览器输入 http://localhost:8080 就可以访问了

# 3. 加载配置文件

## 切换至 press 文件夹
$ cd press 

## 创建 .vuepress 文件夹
$ mkdir .vuepress

## 创建 config.js
$ touch .vuepress/config.js

# 3.1 初始化配置文件

module.export = {
    title: "VuePress 学习",
    description: "VuePress 学习描述"
}

目录结构如下：

.
├── .vuepress
│   └── config.js
└── README.md

# 4. 项目部署

# 4.1 build

$ vuepress build .

该命令执行完毕后会在 .vuepress 文件夹下生成 dist 目录

# 4.2 发布到 github pages 上

  • 创建 github 项目

  • 修改项目名称 （前提项目中必须存在文件，比如上传一个 README.md）

  • 修改 github pages 为下图所示：

# 4.3 在项目根目录创建一个自动执行版本，方便我们每次版本更新

#!/usr/bin/env sh

echo '开始执行命令'
# 生成静态文件
echo '执行命令：vuepress build .'
vuepress build .

# 进入生成的文件夹
echo "执行命令：cd ./.vuepress/dist\n"
cd ./.vuepress/dist

# 初始化一个仓库，仅仅是做了一个初始化的操作，项目里的文件还没有被跟踪
echo "执行命令：git init\n"
git init

# 保存所有的修改
echo "执行命令：git add -A"
git add -A

# 把修改的文件提交
echo "执行命令：commit -m 'deploy'"
git commit -m 'deploy'

# 如果发布到 https://<USERNAME>.github.io
# git push -f git@github.com:<USERNAME>/<USERNAME>.github.io.git master

# 返回到上一次的工作目录
echo "回到刚才工作目录"
cd -

# 4.4 执行上传（前提本地 git 环境与线上已经联通）

bash deploy.sh

# 访问 https://<username>.github.io 即可
